converts Fahrenheit into Celcius *]

function convertx (fahr integer)
{
        return 5 * (fahr -32) / 9;
}

$
    integer     low, high, step;        [* declarations *]
$
    scan (low, high, step);
    while (low <= high )
        {   print (low);
            print (convertx (low));
            low = low + step;
        }
    endwhile
